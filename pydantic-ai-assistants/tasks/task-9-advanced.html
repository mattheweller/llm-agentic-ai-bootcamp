1. Custom Validators

Zod allows custom validation logic beyond basic types. Here's how to create validators for complex business rules:

Custom Validators
import { z } from 'zod';

// Custom email domain validator
const CorporateEmailSchema = z.string()
  .email()
  .refine(
    (email) => email.endsWith('@company.com'),
    { message: 'Must be a company email (@company.com)' }
  );

// Password strength validator
const StrongPasswordSchema = z.string()
  .min(8, 'Password must be at least 8 characters')
  .refine(
    (pwd) => /[A-Z]/.test(pwd),
    { message: 'Must contain at least one uppercase letter' }
  )
  .refine(
    (pwd) => /[0-9]/.test(pwd),
    { message: 'Must contain at least one number' }
  )
  .refine(
    (pwd) => /[!@#$%^&*]/.test(pwd),
    { message: 'Must contain at least one special character' }
  );

// Date range validator
const EventSchema = z.object({
  start_date: z.string().datetime(),
  end_date: z.string().datetime()
}).refine(
  (data) => new Date(data.end_date) > new Date(data.start_date),
  { message: 'End date must be after start date', path: ['end_date'] }
);

2. Error Handling Patterns

Handle validation errors gracefully with detailed, user-friendly messages:

Error Handling
import { z } from 'zod';

const UserSchema = z.object({
  username: z.string().min(3).max(20),
  age: z.number().int().min(18).max(120),
  email: z.string().email()
});

function validateUser(data) {
  try {
    const validated = UserSchema.parse(data);
    return { success: true, data: validated };
  } catch (error) {
    if (error instanceof z.ZodError) {
      // Transform errors into user-friendly format
      const friendlyErrors = error.errors.map(err => ({
        field: err.path.join('.'),
        message: err.message
      }));
      return { success: false, errors: friendlyErrors };
    }
    // Unexpected error
    return { success: false, errors: [{ field: 'unknown', message: error.message }] };
  }
}

// Usage
const result = validateUser({ username: 'ab', age: 15, email: 'invalid' });
if (!result.success) {
  console.log('Validation failed:');
  result.errors.forEach(err => {
    console.log(`  ${err.field}: ${err.message}`);
  });
}

3. Safe Parsing (Non-Throwing)

Use safeParse when you want to handle validation without try-catch:

safeParse vs parse
const schema = z.object({ age: z.number() });

// ❌ parse() throws on failure
try {
  const data = schema.parse({ age: "not-a-number" });
} catch (error) {
  console.log('Validation failed');
}

// ✅ safeParse() returns result object
const result = schema.safeParse({ age: "not-a-number" });

if (result.success) {
  console.log('Valid data:', result.data);
} else {
  console.log('Validation errors:', result.error.errors);
}

4. Retry Logic for API Calls

Implement exponential backoff for transient failures:

Retry with Exponential Backoff
async function retryWithBackoff(fn, maxRetries = 3) {
  for (let attempt = 0; attempt < maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      const isLastAttempt = attempt === maxRetries - 1;

      if (isLastAttempt) {
        throw error;
      }

      // Exponential backoff: 1s, 2s, 4s
      const delay = Math.pow(2, attempt) * 1000;
      console.log(`Attempt ${attempt + 1} failed. Retrying in ${delay}ms...`);
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
}

// Usage with LLM API
async function generateWithRetry(prompt) {
  return retryWithBackoff(async () => {
    const response = await fetch('/api/openai/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ messages: [{ role: 'user', content: prompt }] })
    });

    if (!response.ok) {
      throw new Error(`API error: ${response.status}`);
    }

    return response.json();
  }, 3);
}

5. Discriminated Unions

Handle multiple response types with type safety:

Discriminated Unions
const ResponseSchema = z.discriminatedUnion('status', [
  z.object({
    status: z.literal('success'),
    data: z.object({ id: z.string(), name: z.string() })
  }),
  z.object({
    status: z.literal('error'),
    error_code: z.string(),
    message: z.string()
  }),
  z.object({
    status: z.literal('pending'),
    retry_after: z.number()
  })
]);

// Type-safe handling
function handleResponse(response) {
  const result = ResponseSchema.parse(response);

  switch (result.status) {
    case 'success':
      console.log('Data:', result.data);
      break;
    case 'error':
      console.error(`Error ${result.error_code}: ${result.message}`);
      break;
    case 'pending':
      console.log(`Retry after ${result.retry_after}s`);
      break;
  }
}

6. Transform and Preprocess

Clean and transform data before validation:

Preprocessing
// Trim whitespace and lowercase email
const CleanedEmailSchema = z.string()
  .transform(str => str.trim().toLowerCase())
  .pipe(z.string().email());

// Parse string to number
const StringNumberSchema = z.string()
  .transform(str => parseInt(str, 10))
  .pipe(z.number().int().positive());

// Usage
const email = CleanedEmailSchema.parse('  USER@EXAMPLE.COM  ');
// Returns: 'user@example.com'

const age = StringNumberSchema.parse('25');
// Returns: 25 (number)
